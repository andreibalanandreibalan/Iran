<script>

    // --- CONFIGURARE API ---

    // Rulează 'npm install' în creierul tău și pune cheia de la newsapi.org aici:

    const NEWS_API_KEY = 'PUNE_AICI_CHEIA_TA_DE_LA_NEWSAPI';

    const MONEY_PER_SECOND = 15400; // Estimat John Oliver

   

    let startTime = Date.now();

 

    // 1. FETCH NEWS (Știri reale despre Iran)

    async function fetchNews(query) {

        const feed = document.getElementById('news-feed');

        if (NEWS_API_KEY === 'PUNE_AICI_CHEIA_TA_DE_LA_NEWSAPI') {

            feed.innerHTML = "<p style='color:orange'>[SISTEM]: Introdu un API KEY în cod pentru știri live.</p>";

            return;

        }

 

        try {

            const response = await fetch(`https://newsapi.org/v2/everything?q=${query}&sortBy=publishedAt&apiKey=${NEWS_API_KEY}`);

            const data = await response.json();

           

            feed.innerHTML = '';

            data.articles.slice(0, 5).forEach(art => {

                const div = document.createElement('div');

                div.className = 'news-item';

                div.innerHTML = `<strong>${art.source.name}:</strong> <a href="${art.url}" target="_blank">${art.title}</a>`;

                feed.appendChild(div);

            });

        } catch (e) {

            feed.innerHTML = "Eroare la conectarea cu satelitul de știri.";

        }

    }

 

    // 2. FETCH CURRENCY (Cursul real al Rialului Iranian)

    async function fetchCurrency() {

        try {

            // Folosim un API public gratuit pentru rate de schimb

            const response = await fetch('https://open.er-api.com/v6/latest/USD');

            const data = await response.json();

            const irrRate = data.rates.IRR;

           

            document.getElementById('currency-val').innerText = `${irrRate.toLocaleString()} IRR`;

            document.getElementById('currency-impact').innerText = `1 USD valorează acum atât în Teheran. Inflația ucide mai tăcut decât rachetele.`;

        } catch (e) {

            document.getElementById('currency-val').innerText = "Offline";

        }

    }

 

    // 3. TICKERUL DE BANI (Calcul local)

    function updateMoney() {

        const elapsed = (Date.now() - startTime) / 1000;

        const total = elapsed * MONEY_PER_SECOND;

        document.getElementById('money-counter').innerText = new Intl.NumberFormat('en-US', {

            style: 'currency', currency: 'USD'

        }).format(total);

    }

 

    // Inițializare

    setInterval(updateMoney, 100);

    fetchNews('Iran war');

    fetchCurrency();

</script>
